(i) Testsignal und DFT

x_k = ecg(128);
x_k = x_k(1:4:end);
N = length(x_k)

X_v = fft(x_k);


figure(1);
subplot(2,2,1); plot(x_k); title('x_k'); xlim([1,N]);
subplot(2,2,2); plot(abs(X_v)); title('ABS(FFT(x_k))');xlim([1,N]);

    32
